	write 		--> int ntapfuse_write (const char *path, const char *buf, size_t size, off_t 				off, struct fuse_file_info *fi);

	unlink 		--> int ntapfuse_unlink (const char *path);

	mkdir 		--> int ntapfuse_mkdir (const char *path, mode_t mode);

	rmdir 		--> int ntapfuse_rmdir (const char *path);

	truncate	--> int ntapfuse_truncate (const char *path, off_t off);

	chown 		--> int ntapfuse_chown (const char *path, uid_t uid, gid_t gid);

	link 		--> int ntapfuse_link (const char *src, const char *dst);
